import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the dataset
data = pd.read_csv('data.csv')

# Split the dataset into features (X) and labels (y)
features = data[['Feature1', 'Feature2']]
labels = data['Label']

# Split the dataset into training and testing subsets
def train_test_split_custom(X, y, test_size=0.2, random_state=None):
    if random_state:
        np.random.seed(random_state)
    indices = np.random.permutation(len(X))
    test_size = int(len(X) * test_size)
    test_indices = indices[:test_size]
    train_indices = indices[test_size:]
    X_train, X_test = X[train_indices], X[test_indices]
    y_train, y_test = y[train_indices], y[test_indices]
    return X_train, X_test, y_train, y_test

features_train, features_test, labels_train, labels_test = train_test_split_custom(features.values, labels.values, test_size=0.2, random_state=42)

# Define the logistic function
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# Initialize weights and bias
np.random.seed(42)
weights = np.random.rand(features_train.shape[1])
bias = np.random.rand()

# Train the logistic regression model using gradient descent
learning_rate = 0.01
epochs = 1000
losses = []  # Store the loss at each iteration

for epoch in range(epochs):
    # Forward pass
    predictions = sigmoid(np.dot(features_train, weights) + bias)
    
    # Compute loss
    error = labels_train - predictions
    loss = -np.mean(labels_train * np.log(predictions) + (1 - labels_train) * np.log(1 - predictions))
    losses.append(loss)
    
    # Backpropagation
    weights += learning_rate * np.dot(features_train.T, error)
    bias += learning_rate * np.sum(error)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score
from sklearn.preprocessing import PolynomialFeatures

# Step 1: Load the dataset
data = pd.read_csv('data.csv')

# Step 2: Feature Engineering - Adding new features
poly = PolynomialFeatures(degree=2, interaction_only=True)
enhanced_features = poly.fit_transform(data[['Feature1', 'Feature2']])
enhanced_df = pd.DataFrame(enhanced_features, columns=poly.get_feature_names_out(['Feature1', 'Feature2']))

# Step 3: Split the dataset into training and testing subsets
X_train, X_test, y_train, y_test = train_test_split(enhanced_df, data['Label'], test_size=0.2, random_state=42)

# Step 4: Train a logistic regression model
model = LogisticRegression()
model.fit(X_train, y_train)

# Step 5: Plot the data points and decision boundary
def plot_decision_boundary(X, y, model):
    h = .02
    x_min, x_max = X[:, 1].min() - 1, X[:, 1].max() + 1
    y_min, y_max = X[:, 2].min() - 1, X[:, 2].max() + 1
    xx, yy = np.meshgrid(np.arange(x_min, x_max, h), np.arange(y_min, y_max, h))
    Z = model.predict(poly.fit_transform(np.c_[xx.ravel(), yy.ravel()]))
    Z = Z.reshape(xx.shape)
    plt.contourf(xx, yy, Z, cmap=plt.cm.RdBu, alpha=0.8)
    plt.scatter(X[:, 1], X[:, 2], c=y, cmap=plt.cm.RdBu, edgecolors='k')
    plt.xlabel('Feature1')
    plt.ylabel('Feature2')
    plt.title('Logistic Regression Decision Boundary')

plot_decision_boundary(X_train.values, y_train, model)
plt.show()

# Step 6: Evaluate the model on testing data
predictions = model.predict(X_test)
accuracy = accuracy_score(y_test, predictions)
print("Accuracy:", accuracy)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Step 1: Load the dataset
data = pd.read_csv('data.csv')

# Step 2: Feature Engineering - Adding new features
data['Feature1_squared'] = data['Feature1'] ** 2
data['Feature1_times_Feature2'] = data['Feature1'] * data['Feature2']

# Step 3: Split the dataset into features (X) and labels (y)
X = data[['Feature1', 'Feature2', 'Feature1_squared', 'Feature1_times_Feature2']]
y = data['Label']

# Step 4: Split the dataset into training and testing subsets
def train_test_split_custom(X, y, test_size=0.2, random_state=None):
    np.random.seed(random_state)
    shuffled_indices = np.random.permutation(len(X))
    test_set_size = int(len(X) * test_size)
    test_indices = shuffled_indices[:test_set_size]
    train_indices = shuffled_indices[test_set_size:]
    return X.iloc[train_indices], X.iloc[test_indices], y.iloc[train_indices], y.iloc[test_indices]

X_train, X_test, y_train, y_test = train_test_split_custom(X, y, test_size=0.2, random_state=42)

# Step 5: Define sigmoid function
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# Step 6: Initialize weights and bias
np.random.seed(42)
weights = np.random.randn(X_train.shape[1])
bias = np.random.randn()

# Step 7: Train the model using gradient descent
learning_rate = 0.01
num_epochs = 1000

for epoch in range(num_epochs):
    # Forward pass
    linear_combination = np.dot(X_train, weights) + bias
    predictions = sigmoid(linear_combination)
    
    # Compute loss
    loss = -np.mean(y_train * np.log(predictions) + (1 - y_train) * np.log(1 - predictions))
    
    # Backward pass
    dw = np.dot(X_train.T, (predictions - y_train)) / len(X_train)
    db = np.mean(predictions - y_train)
    
    # Update weights and bias
    weights -= learning_rate * dw
    bias -= learning_rate * db
    
    # Print loss every 100 epochs
    if epoch % 100 == 0:
        print(f'Epoch {epoch}, Loss: {loss}')

# Step 8: Make predictions on the testing data
test_predictions = sigmoid(np.dot(X_test, weights) + bias)

# Step 9: Plot data points
plt.scatter(data['Feature1'], data['Feature2'], c=data['Label'], cmap='coolwarm', edgecolors='k', label='Data Points')

# Step 10: Plot decision boundary
x_min, x_max = data['Feature1'].min() - 1, data['Feature1'].max() + 1
y_min, y_max = data['Feature2'].min() - 1, data['Feature2'].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, 0.01), np.arange(y_min, y_max, 0.01))
Z = sigmoid(weights[0] * xx + weights[1] * yy + weights[2] * (xx ** 2) + weights[3] * (xx * yy) + bias)
plt.contourf(xx, yy, Z, alpha=0.3, cmap='coolwarm')
plt.xlabel('Feature1')
plt.ylabel('Feature2')
plt.title('Logistic Regression Decision Boundary')
plt.legend()
plt.show()

# Step 11: Evaluate the model on the testing data
accuracy = accuracy_score(y_test, test_predictions.round())
print("Accuracy:", accuracy)
